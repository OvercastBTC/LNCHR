	stringsplit, vers, whatversionstring, .  ;<---- there is a period there
	if(vers0 < incwhatpart){
		vers0 := incwhatpart
	}
	
	vers%incwhatpart%++

	loop, %vers0%{
		thisversion := vers%a_index%
		if(thisversion = ""){
			thisversion := "0"
		}
		newversionstring := newversionstring . thisversion
		if(a_index < vers0){
			newversionstring := newversionstring . "."

	}

}
